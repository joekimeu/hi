            // Paginate successor jobs in the tree so only 5 show at a time (per level)
            function initSuccessorPagination() {
              $('ul.successor-records').each(function () {
                const $list = $(this);
                const pageSize = parseInt($list.data('page-size'), 10) || 5;
                const $items = $list.children('li');
                const totalItems = $items.length;
                const totalPages = Math.max(1, Math.ceil(totalItems / pageSize));

                // The pagination controls are inserted immediately after the list
                const $pager = $list.next('.successor-pagination');
                const $prev = $pager.find('.successor-prev');
                const $next = $pager.find('.successor-next');
                const $info = $pager.find('.successor-page-info');

                // Nothing to paginate
                if (totalItems <= pageSize) {
                  $pager.hide();
                  return;
                }

                $list.data('current-page', 1);
                $list.data('page-size', pageSize);
                $list.data('total-pages', totalPages);

                function render(page) {
                  const clamped = Math.min(Math.max(page, 1), totalPages);
                  const start = (clamped - 1) * pageSize;
                  const end = start + pageSize;

                  $items.each(function (idx) {
                    $(this).toggle(idx >= start && idx < end);
                  });

                  $info.text(`Page ${clamped} of ${totalPages}`);
                  $prev.prop('disabled', clamped === 1);
                  $next.prop('disabled', clamped === totalPages);
                  $list.data('current-page', clamped);
                }

                $prev.off('click').on('click', function () {
                  render(($list.data('current-page') || 1) - 1);
                });
                $next.off('click').on('click', function () {
                  render(($list.data('current-page') || 1) + 1);
                });

                // Initial render
                render(1);
              });
            }
